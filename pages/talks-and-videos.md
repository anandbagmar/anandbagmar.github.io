<script>
document.addEventListener('DOMContentLoaded', function () {
  document.querySelectorAll('.eot-featured-card').forEach(function (card) {
    card.addEventListener('click', function (e) {
      if (e.target.closest('.eot-featured-close')) return;
      var videoId = card.dataset.videoId;
      var thumb = card.querySelector('.eot-featured-thumb');
      if (!thumb || card.classList.contains('is-playing')) return;

      // Replace thumb with embed
      var embed = document.createElement('div');
      embed.className = 'eot-featured-embed';
      embed.innerHTML =
        '<iframe src="https://www.youtube.com/embed/' + videoId +
        '?autoplay=1&rel=0" allow="autoplay; encrypted-media" allowfullscreen></iframe>' +
        '<button class="eot-featured-close" title="Close">✕</button>';
      thumb.replaceWith(embed);
      card.classList.add('is-playing');

      // Close button
      embed.querySelector('.eot-featured-close').addEventListener('click', function (ev) {
        ev.stopPropagation();
        var newThumb = document.createElement('div');
        newThumb.className = 'eot-featured-thumb';
        newThumb.innerHTML =
          '<img src="https://img.youtube.com/vi/' + videoId +
          '/mqdefault.jpg" loading="lazy" />' +
          '<div class="eot-featured-play">▶</div>';
        embed.replaceWith(newThumb);
        card.classList.remove('is-playing');
      });
    });
  });
});
</script>
